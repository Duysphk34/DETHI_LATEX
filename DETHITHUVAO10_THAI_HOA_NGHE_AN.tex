	\begin{enumerate}
		\item Chứng minh $OFMC$ là tứ giác nội tiếp.
		\item Gọi $K$ là giao điểm của hai đường thẳng $CM$ và $AB$. Chứng minh $\triangle CMF\sim \triangle EMA$ và $KF. OA=AF. KB$.

	\end{enumerate}
	\loigiai{}
\end{bt}
