\RenewEnviron{listEX}[1][]{
	\ifthenelse{\equal{#1}{1}}%
	{%
		\begin{enumerate}%
			\BODY%
		\end{enumerate}%
	}%
	{\begin{multicols}{#1}
			\begin{enumerate}%
				\BODY%
			\end{enumerate}%
	\end{multicols}}%
}

\RenewEnviron{enumEX}[2][]{

	}%
	{\begin{multicols}{#2}
			\begin{enumerate}%
				\BODY%
			\end{enumerate}%
	\end{multicols}}%
}
