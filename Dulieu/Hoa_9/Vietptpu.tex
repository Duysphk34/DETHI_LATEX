\begin{ex}[Cân bằng phản ứng oxi hóa - khử] 
	Cân bằng các phương trình phản ứng sau bằng phương pháp thăng bằng electron
	\begin{enumerate}[(1)]
		\item$KMnO_4$ \MuiTen[$t^{\circ}$] $K_2MnO_4$ \explus $MnO_2$ \explus $O_2$
		\item$2Ca$ \explus $O_2$ \MuiTen $2CaO$
		\item$A3$ \explus $B3$ \explus $C3$ \MuiTen $X3$ \explus $Y3$ \explus $Z3$
		\item$Aa1$ \explus $Bb1$ \explus $Cc1$ \MuiTen $Xa1$ \explus $Ya1$
		\item$Aa2$ \explus $Bb2$ \explus $Cc2$ \MuiTen $Xa2$ \explus $Ya2$ \explus $Za2$
		\item$Aa3$ \explus $Bb3$ \explus $Cc3$ \MuiTen $Xa3$ \explus $Ya3$ \explus $Za3$
	\end{enumerate}
\end{ex}
