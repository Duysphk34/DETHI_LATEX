	\ifblank{#4}{\def\tuychonfour{black}}{\def\tuychonfour{#4}}
	\ignorespaces\hspace*{-14pt}\canhdong{\begin{tikzpicture}[declare function={dodai=\tuychonone mm;goc=90;}]

	\end{tikzpicture}}%
}
