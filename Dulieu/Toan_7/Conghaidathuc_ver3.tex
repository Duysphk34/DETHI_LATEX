 Hãy tính:

	\item $A(x)+B(x)$.
	\item $A(x)-B(x)$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $A(x)+B(x)$ 
	\begin{eqnarray*}
	\text{Ta có}\ A(x)+B(x)
	&=&\left(-9x+5x^4+4x^3-2x^2+7+3x^5\right)+\left(9x^3+2x^4+8x^5-4+2x^2+3x\right)\\
	&=&\left(3 x^5 + 5 x^4 + 4 x^3 - 2 x^2 - 9 x + 7\right)+\left(8 x^5 + 2 x^4 + 9 x^3 + 2 x^2 + 3 x - 4\right)\\
	&=&(3+8)x^5+(5+2)x^4+(4+9)x^3+(-2+2)x^2+(-9+3)x+(7-4)\\
	&=&11 x^5 + 7 x^4 + 13 x^3 - 6 x + 3
	\end{eqnarray*}
	\item $A(x)-B(x)$ 
	\begin{eqnarray*}
	\text{Ta có}\ A(x)-B(x)
	&=&\left(-9x+5x^4+4x^3-2x^2+7+3x^5\right)-\left(9x^3+2x^4+8x^5-4+2x^2+3x\right)\\

	&=&(3-8)x^5+(5-2)x^4+(4-9)x^3+(-2-2)x^2+(-9-3)x+(7+4)\\
	&=&- 5 x^5 + 3 x^4 - 5 x^3 - 4 x^2 - 12 x + 11
	\end{eqnarray*}
\end{enumerate}
}
\end{bt}
