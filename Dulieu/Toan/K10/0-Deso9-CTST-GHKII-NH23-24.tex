	Xét tính đúng, sai của các khẳng định sau
	\choiceTF
	{\True $x^2+4 x+3<0$ khi $x \in(-3 ;-1)$}
	{\True $x^2-6 x+8 \geq 0$ khi $x \in(-\infty ; 2] \cup[4 ;+\infty)$}
	{$f(x)=x^2-x+5$ luôn âm với mọi $x$ thuộc $\mathbb{R}$}
	{\True $f(x)=-36 x^2+12 x-1$ luôn nhỏ hơn hoặc bằng $0$ với mọi $x \in \mathbb{R}$}
	\loigiai{
		\begin{itemize}
			\item $x^2+4 x+3<0$ khi $x \in(-3 ;-1)$.\\
			Xét $f(x)=x^2+4x+3$ có $\Delta=1>0$, $a=1>0$ và có hai nghiệm $x_1=-3$; $x_2=-1$.\\
			Do đó, ta có bảng xét dấu sau
			\begin{center}
				\begin{tikzpicture}
					\tkzTabInit[nocadre=false,lgt=2,espcl=2,deltacl=0.5]
					{$x$/1 , $f(x)$/1}
					{$-\infty$,$-3$,$-1$,$+\infty$}
					\tkzTabLine{,+,z,-,z,+}
				\end{tikzpicture}	
			\end{center}
			Suy ra $f(x)=x^2+4 x+3<0$ khi $x\in (-3;-1)$.\\
			Vậy tập nghiệm của bất phương trình $S=(-3 ;-1)$.
			\item $x^2-6x+8 \geq 0$ khi $x \in(-\infty ; 2] \cup[4 ;+\infty)$.\\
			Xét $f(x)=x^2-6x+8$ có $\Delta'=1>0$, $a=1>0$ và có hai nghiệm $x_1=2$; $x_2=4$.\\
			Do đó, ta có bảng xét dấu sau
			\begin{center}
				\begin{tikzpicture}
					\tkzTabInit[nocadre=false,lgt=2,espcl=2,deltacl=0.5]
					{$x$/1 , $f(x)$/1}
					{$-\infty$,$2$,$4$,$+\infty$}
					\tkzTabLine{,+,z,-,z,+}
				\end{tikzpicture}	
			\end{center}
			Suy ra $f(x)=x^2-6x+8 \ge 0$ khi $x \in(-\infty ; 2]\cup[4 ;+\infty)$.\\
			Vậy tập nghiệm của bất phương trình là $S=(-\infty ; 2] \cup[4 ;+\infty)$.
			\item $f(x)=x^2-x+5$ luôn âm với mọi $x$ thuộc $\mathbb{R}$.\\
			Ta có $f(x)=x^2-x+5=x^2-2\cdot x \cdot \dfrac{1}{2}+\dfrac{1}{4}-\dfrac{1}{4}+5=\left(x-\dfrac{1}{2}\right)^2+\dfrac{19}{4} \ge \dfrac{19}{4}$, $\forall x \in \mathbb{R}$.\\ Vì vậy, $f(x)>0$, $\forall x \in \mathbb{R}$.
			\item $f(x)=-36 x^2+12 x-1$ luôn nhỏ hơn hoặc bằng $0$ với mọi $x \in \mathbb{R}$.\\
			Ta có $f(x)=-36x^2+12x-1=-\left[(6x)^2-2\cdot 6 x+1\right]=-(6x-1)^2 \le 0$, $\forall x \in \mathbb{R}$.
		\end{itemize}
	}
\end{ex}
\begin{ex}%[0D7H2-1]%[Dự án đề kiểm tra Toán khối 10 GHKII NH23-24-Dot 2-Võ Thị Thùy Trang]%[Deso 9-CTST]
	Cho $2$ phương trình $\sqrt{5 x+10}=8-x \quad(1)$ và $\sqrt{3 x^2-9 x+1}=x-2 \quad(2)$. Khi đó
	\choiceTF
	{\True Phương trình ($1$) có $1$ nghiệm}
	{Phương trình ($2$) có $2$ nghiệm}
	{\True Phương trình ($1$) và ($2$) có chung tập nghiệm}
	{\True Tổng các nghiệm của phương trình ($1$) và ($2$) bằng $6$}
	\loigiai{
		\begin{itemize}
			\item ($1$)$\colon \sqrt{5 x+10}=8-x$.\\
			Cách giải $1\colon$\\
			Bình phương hai vế phương trình, ta được
			\allowdisplaybreaks
			\begin{eqnarray*}
				5x+10=64-16x+x^2 &\Leftrightarrow& x^2-21x+54=0\\
				&\Leftrightarrow& \hoac{&x=3\\&x=18.}
			\end{eqnarray*}
			Thay $x=3$ vào phương trình đã cho ta được $\sqrt{25}=5$ (thỏa mãn).\\
			Thay $x=18$ vào phương trình đã cho ta được $\sqrt{100}=-10$ (không thỏa mãn).\\
			Vậy tập nghiệm phương trình là $S=\{3\}$.\\
			Cách giải $2\colon$ Ta có\\
			\begin{eqnarray*}
				\sqrt{5x+10}=8-x &\Leftrightarrow& \heva{&8-x \ge 0\\& 5x+10=64-16 x+x^2}\\
				&\Leftrightarrow &\heva{&x\le 8\\&x^2-21x+54=0}\\
				&\Leftrightarrow&\heva{&x\le 8 \\& x=3 \vee x=18}\Leftrightarrow x=3
			\end{eqnarray*}
			Vậy tập nghiệm phương trình là $S=\{3\}$.
			\item ($2$)$\colon \sqrt{3 x^2-9 x+1}=x-2$.\\
			Cách giải $1\colon$\\
			Bình phương hai vế phương trình, ta được
			$$
			3x^2-9x+1=x^2-4x+4 \Leftrightarrow 2x^2-5x-3=0 \Leftrightarrow x=3 \vee x=-\dfrac{1}{2}.
			$$
			Thay $x=3$ vào phương trình đã cho, ta được $\sqrt{1}=1$ (thỏa mãn).\\
			Thay $x=-\dfrac{1}{2}$ vào phương trình đã cho, ta được $\sqrt{\dfrac{25}{4}}=-\dfrac{5}{2}$ (không thỏa mãn).\\ Vậy tập nghiệm phương trình là $S=\{3\}$.\\
			Cách giải $2\colon$ Ta có
			$$
			\Leftrightarrow\heva{&x \ge 2\\&2x^2-5x+3=0}
			\Leftrightarrow \heva{&x\ge 2\\&x=3 \vee x=-\dfrac{1}{2}}
			\Leftrightarrow x=3.
			$$	
			Vậy tập nghiệm phương trình là $S=\{3\}$.
		\end{itemize}
		
	}			
\end{ex}

\begin{ex}%[0H9V4-2]%[Dự án đề kiểm tra Toán khối 10 GHKII NH23-24-Dot 2-Võ Thị Thùy Trang]%[Deso 9-CTST]
	Đường tròn $(C)$ đi qua hai điểm $A(1;2)$, $B(3;4)$ và tiếp xúc $\Delta\colon 3x+y-3=0$. Khi đó
	\choiceTF
	{\True Có hai đường tròn $(C)$ thỏa mãn}
	{Tổng đường kính của các đường tròn $(C)$ bằng $2\sqrt{10}$}
	{\True Điểm $M(3;2)$ nằm bên trong của đường tròn $(C)$}
	{\True Điểm $N(1;0)$ nằm trên ít nhất một đường tròn $(C)$}
	\loigiai{
		\begin{itemize}
			\item 
			Gọi tâm đường tròn là $I(a;b)$, ta có $\mathrm{d}(I,\Delta)=\dfrac{\left|3a+b-3\right|}{\sqrt{10}}$.\\
			Theo giả thiết 
			\begin{eqnarray*}
				\heva{&IA^2=IB^2\\&IA^2=(\mathrm{d}(I, \Delta))^2}
				&\Leftrightarrow & \heva{&(a-1)^2+(b-2)^2=(a-3)^2+(b-4)^2\\ &(a-1)^2+(b-2)^2=\dfrac{(3a+b-3)^2}{10}}\\
				&\Leftrightarrow &\heva{&a+b=5 \quad(1)\\&a^2-2a+9b^2-34b+41-6ab=0\quad(2)}
			\end{eqnarray*}
			Thay ($1$) vào ($2$) ta được\\
			$(5-b)^2-2(5-b)+9b^2-34 b+41-6(5-b)b=0$
			$\Leftrightarrow 4b^2-18b+14=0 \Leftrightarrow\hoac{&b=1\\ &b=\dfrac{7}{2}.}$\\
			Với $b=1$ ta có $a=4$, $R=\sqrt{10}$.\\
			Với $b=\dfrac{7}{2}$ ta có $a=\dfrac{3}{2}$, $R=\dfrac{\sqrt{10}}{2}$.\\
			Vậy có hai đường tròn thỏa mãn $\left(x-\dfrac{7}{2}\right)^2+\left(y-\dfrac{3}{2}\right)^2=\dfrac{5}{2}$ và $(x-4)^2+(y-1)^2=10$.
			\item	Tổng đường kính của các đường tròn $(C)$ bằng $3\sqrt{10}$.

	Cho bất phương trình $\left(m^2-4\right) x^2+(m-2) x+1<0$. Tập tất cả các giá trị của tham số $m$ làm cho bất phương trình vô nghiệm có dạng $(-\infty ; a] \cup[b ;+\infty)$. Tính giá trị của $a \cdot b$.
	\loigiai{Xét bất phương trình 
		$$\left(m^2-4\right) x^2+(m-2) x+1<0.\quad\quad(1)$$
		Truờng hợp 1: $ m^2-4=0\Leftrightarrow \hoac{&m=2 \\ &m=-2}$
		\begin{itemize}
			\item Với $m=2$ thì (1) trở thành $1<0$, vô nghiệm. Do đó, $m=2$ thỏa yêu cầu bài toán.
			\item Với $m=-2$ thì (1) trở thành $-4x+1<0\Leftrightarrow x>\dfrac{1}{4}$. Do đó, $m=-2$ không thỏa yêu cầu bài toán.
		\end{itemize}
		Truờng hợp 2: $m\ne \pm 2$. Bất phương trình (1) vô nghiệm khi và chỉ khi 
		\begin{eqnarray*}
			&&\left(m^2-4\right) x^2+(m-2) x+1 \geq 0, \quad \forall x \in \mathbb{R}\\
			&\Leftrightarrow& \heva{&a=m^2-4>0 \\ &\Delta=(m-2)^2-4\left(m^2-4\right) \leq 0}\\ &\Leftrightarrow&\heva{&\hoac{&m>2 \\ &m<-2} \\ &\hoac{&m \leq-\dfrac{10}{3} \\ &m \geq 2} 
			}\\
			&\Leftrightarrow&\hoac{&m \leq-\dfrac{10}{3}, \\ &m>2.}
		\end{eqnarray*}
		Từ hai trường hợp trên ta có $m \in\left(-\infty ;-\dfrac{10}{3}\right] \cup[2 ;+\infty)$. Vậy $a \cdot b=-\dfrac{20}{3}$.}
