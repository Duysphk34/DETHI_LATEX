\begin{bt}
Cho hai đa thức $A(x) =8 x^4 - 8 x^3 + 5 x - 3$ và $B(x)=8 x^5 - 2 x^4 + 9 x^3 + 7 x + 3$
 Hãy tính:
	\begin{enumerate}
	\item $A(x)+B(x)$.
	\item $A(x)-B(x)$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $A(x)+B(x)$ 
	\begin{eqnarray*}
	\text{Ta có}\ A(x)+B(x)
	&=&\left(8 x^4 - 8 x^3 + 5 x - 3\right)+\left(8 x^5 - 2 x^4 + 9 x^3 + 7 x + 3\right)\\
	&=&8x^5+(8-2)x^4+(-8+9)x^3+(5+7)x+(-3+3)\\
	&=&8 x^5 + 6 x^4 + x^3 + 12 x
	\end{eqnarray*}
	\item $A(x)-B(x)$ 
	\begin{eqnarray*}
	\text{Ta có}\ A(x)-B(x)
	&=&\left(8 x^4 - 8 x^3 + 5 x - 3\right)-\left(8 x^5 - 2 x^4 + 9 x^3 + 7 x + 3\right)\\
	&=&-8x^5+(8+2)x^4+(-8-9)x^3+(5-7)x+(-3-3)\\
	&=&- 8 x^5 + 10 x^4 - 17 x^3 - 2 x - 6
	\end{eqnarray*}
\end{enumerate}
}
\end{bt}
